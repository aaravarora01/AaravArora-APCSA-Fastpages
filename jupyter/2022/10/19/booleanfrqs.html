public int getChargeStartTime(int chargeTime) {
    int startTime = 0;
    for (int i = 1; i < 24; i++) {
        if (this.getChargingCost(i, chargeTime)
            this.getChargingCost(startTime, chargeTime)) {
            startTime = i;
        }
    }
    return startTime;
}
